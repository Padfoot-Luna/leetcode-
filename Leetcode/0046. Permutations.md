`Middle`

> Given a collection of **distinct** integers, return all possible permutations.
>
> 给定一个 **没有重复** 数字的序列，返回其所有可能的全排列。

#### 1. Backtracking DFS

```python
class Solution(object):
    def permute(self, nums):
        if len(nums) <= 1: return [nums]
        visited = [False for _ in nums]  # 记录那些元素已经访问
        ans = []

        def dfs(numbers, result, path, visited):
            if len(path) == len(numbers):  
                result.append(path[:])  # 这里记得用path[:]或拷贝
                return
            for i in range(len(numbers)):
                if visited[i]:  # 如果已经访问过某元素，直接跳过进下一个元素
                    continue
                path.append(numbers[i])
                visited[i] = True  # 将访问过的元素标记
                dfs(numbers, result, path, visited)
                path.pop()  # 恢复到之前状态
                visited[i] = False  # 恢复到之前状态

        dfs(nums, ans, [], visited)
        return ans
```

```python
class Solution(object):
    def permute(self, nums):
        if len(nums) <= 1: return [nums]
        visited = [False for _ in nums]  # 记录那些元素已经访问
        ans = []
        n = len(nums)
        
        def dfs(path):
            if len(path) == n:  
                ans.append(path[:])  # 这里记得用path[:]或拷贝
                return
            for i in range(n):
                if visited[i]:  # 如果已经访问过某元素，直接跳过进下一个元素
                    continue
                path.append(nums[i])
                visited[i] = True  # 将访问过的元素标记
                dfs(path)
                path.pop()  # 恢复到之前状态
                visited[i] = False  # 恢复到之前状态

        dfs([])
        return ans
```

```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        ans = []
        
        def backtrack(path, rest):
            if not rest:
                ans.append(path)
            for i in range(len(rest)):
                backtrack(path+[rest[i]], rest[:i]+rest[i+1:])   # 切片不担心越界报错
        backtrack([], nums)
        return ans
```

#### 2. 递归 BSF

```python
class Solution:
    def permute(self, nums: List[int]) -> List[List[int]]:
        if len(nums) <= 1: return [nums]
        ans = []
        for i, x in enumerate(nums):  
            new_nums = nums[:i] + nums[i+1:]
            for r in self.permute(new_nums):
                ans.append([x] + r)
        return ans
```
